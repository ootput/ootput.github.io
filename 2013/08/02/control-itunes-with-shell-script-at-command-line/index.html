Control iTunes with Shell Script at Command-line

Use the following script to control iTunes from the shell:

[sourcecode]

!/bin/sh

#

#

iTunes Command Line Control v1.0.0.1

written by David Schlosnagle

created 2001.11.08

modified 2007.01.28 by David West

modified 2013.08.02 by ootput

filename=itunes

#

showHelp () {
echo "——————————————-";
echo "iTunes Command Line Interface";
echo "——————————————-";
echo "Usage: basename $0 ";
echo;
echo "Options:";
echo " status = Shows iTunes’ status, current artist and track.";
echo " play = Start playing iTunes.";
echo " pause = Pause iTunes.";
echo " next = Go to the next track.";
echo " prev = Go to the previous track.";
echo " mute = Mute iTunes’ volume.";
echo " unmute = Unmute iTunes’ volume.";
echo " vol up = Increase iTunes’ volume by 10%";
echo " vol down = Increase iTunes’ volume by 10%";
echo " vol # = Set iTunes’ volume to # [0-100]";
echo " stop = Stop iTunes.";
echo " search = Run query, populate playlist named foo and play. "
echo " playlist = Show playlists saved in iTunes.";
echo " clear = clear playlist foo. ";
echo " quit = Quit iTunes.";
}

if [ $# = 0 ]; then
showHelp;
fi

while [ $# -gt 0 ]; do
arg=$1;
songname=$2;
echo $songname;
case $arg in
"status" ) state=osascript -e 'tell application &quot;iTunes&quot; to player state as string';
echo "iTunes is currently $state.";
if [ $state = "playing" ]; then
artist=osascript -e 'tell application &quot;iTunes&quot; to artist of current track as string';
track=osascript -e 'tell application &quot;iTunes&quot; to name of current track as string';
echo "Current track $artist: $track";
fi
break ;;

    &quot;play&quot; ) echo &quot;Playing iTunes.&quot;;
        osascript -e 'tell application &quot;iTunes&quot; to play';
        break ;;

    &quot;pause&quot; ) echo &quot;Pausing iTunes.&quot;;
        osascript -e 'tell application &quot;iTunes&quot; to pause';
        break ;;

    &quot;next&quot; ) echo &quot;Going to next track.&quot; ;
        osascript -e 'tell application &quot;iTunes&quot; to next track';
        break ;;

    &quot;prev&quot; ) echo &quot;Going to previous track.&quot;;
        osascript -e 'tell application &quot;iTunes&quot; to previous track';
        break ;;

    &quot;mute&quot; ) echo &quot;Muting iTunes volume level.&quot;;
        osascript -e 'tell application &quot;iTunes&quot; to set mute to true';
        break ;;

    &quot;unmute&quot; ) echo &quot;Unmuting iTunes volume level.&quot;;
        osascript -e 'tell application &quot;iTunes&quot; to set mute to false';
        break ;;

    &quot;vol&quot; ) echo &quot;Changing iTunes volume level.&quot;;
        vol=`osascript -e 'tell application &quot;iTunes&quot; to sound volume as integer'`;
        if [ $2 = &quot;up&quot; ]; then
            newvol=$(( vol+10 ));
        fi

        if [ $2 = &quot;down&quot; ]; then
            newvol=$(( vol-10 ));
        fi

        if [ $2 -gt 0 ]; then
            newvol=$2;
        fi
        osascript -e &quot;tell application &quot;iTunes&quot; to set sound volume to $newvol&quot;;
        break ;;

    &quot;stop&quot; ) echo &quot;Stopping iTunes.&quot;;
        osascript -e 'tell application &quot;iTunes&quot; to stop';
        break ;;

    &quot;quit&quot; ) echo &quot;Quitting iTunes.&quot;;
        osascript -e 'tell application &quot;iTunes&quot; to quit';
        exit 1 ;;

    &quot;search&quot; ) echo &quot;Searching Library.&quot;;
        songname=$2;
        osascript -e &quot;tell application &quot;iTunes&quot;&quot; -e &quot;set searchResults to search playlist &quot;Library&quot; for &quot;$songname&quot;&quot; -e &quot;repeat with aTrack in searchResults&quot; -e &quot;copy aTrack to playlist &quot;foo&quot;&quot; -e &quot;end repeat&quot; -e &quot;play playlist &quot;foo&quot;&quot; -e &quot;end tell&quot;;
        break ;;

    &quot;playlist&quot; )
        if [ -n &quot;$2&quot; ]; then
            echo &quot;Changing iTunes playlists to '$2'.&quot;;
            osascript -e 'tell application &quot;iTunes&quot;' -e &quot;set new_playlist to \&quot;$2\&quot; as string&quot; -e &quot;play playlist new_playlist&quot; -e &quot;end tell&quot;;
            break ;
        else
            # Show available iTunes playlists.
            echo &quot;Playlists:&quot;;
            osascript -e 'tell application &quot;iTunes&quot;' -e &quot;set allPlaylists to (get name of every playlist)&quot; -e &quot;end tell&quot;;
            break;
        fi
        break;;

    &quot;clear&quot; ) echo &quot;Clearing Query.&quot;;
        osascript -e &quot;tell application &quot;iTunes&quot; to delete tracks of playlist &quot;foo&quot;&quot;;
        break ;;

    &quot;help&quot; | * ) echo &quot;help:&quot;;
        showHelp;
        break ;;
esac

done
[/sourcecode]
